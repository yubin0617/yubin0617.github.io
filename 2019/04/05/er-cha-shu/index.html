 建立一棵二叉树，试编程实现二叉树的如下基本操作
  1. 按先序序列构造一棵二叉链表表示的二叉树T；
  2. 对这棵二叉树进行遍历：先序、中序、后序以及层次遍历，分别输出结点的遍历序列；
  3. 求二叉树的深度/结点数目/叶结点数目；(选做)

  4. 将二叉树每个结点的左右子树交换位置。（选做）
    <–!more–>
    `
    #include
    using namespace std;
    //定义节点
    typedef struct node //定义二叉树存储结构
    {
    struct node lchild;
    struct node     rchild;
    char data;
    }BiTreeNode, BiTree;
    typedef struct //定义队列
    {
    BiTree     base;
    int front;
    int rear;
    }Squeue;
    int chushihua(Squeue &Q) //出栈函数
    {
    Q.base = new BiTree[100]; //分配栈的空间
    if (!Q.base) //判断是否分配成功

     return 0;

    Q.front = Q.rear = 0; //队列初始化为空
    return 0;
    }
    int rudui(Squeue &Q,BiTree T) //出队函数
    {
    if ((Q.rear + 1) % 100 == Q.front) //如果队满，就退出

     return -1;

    Q.base[Q.rear] = T; //入队
    Q.rear = (Q.rear + 1) % 100; //队列的顶加一
    return 0;
    }
    int chudui(Squeue &Q, BiTree &S) //出队
    {
    if (Q.front == Q.rear) //判断队列是否为空

     return -1;

    S= Q.base[Q.front]; //s出队
    Q.front = (Q.front + 1) % 100; //队尾指针加一
    return 0;
    }

                                            //按照前序顺序建立二叉树

    void createBiTree(BiTree &T) //&的意思是传进来节点指针的引用，括号内等价于 BiTreeNode* &T,目的是让传递进来的指针发生改变
    {
    char c;
    cin >> c;
    if (‘#’ == c) //当遇到#时，令树的根节点为NULL，从而结束该分支的递归

     T = NULL;

    else
    {

     T = new BiTreeNode;                   //创建新节点树
 T->data = c;                          //新节点为c
 createBiTree(T->lchild);              //创建左子树
 createBiTree(T->rchild);              //创建右子树

    }
    }

                                           //前序遍历二叉树并打印

    void preTraverse(BiTree T)
    {
    if (T)
    {

     cout << T->data << " ";              //输出中间节点
 preTraverse(T->lchild);              //遍历左子树
 preTraverse(T->rchild);              //遍历右子树

    }
    }

                                          //中序遍历二叉树并打印

    void midTraverse(BiTree T)
    {
    if (T)
    {

     midTraverse(T->lchild);              //遍历左子树
 cout << T->data << " ";              //遍历输出
 midTraverse(T->rchild);              //遍历右子树

    }
    }

                                         //后续遍历二叉树并打印

    void postTraverse(BiTree T)
    {
    if (T)
    {

     postTraverse(T->lchild);           //遍历左子树             
 postTraverse(T->rchild);           //遍历右子树
 cout << T->data << " ";            //输出节点

    }
    }

                                        //层序遍历二叉树并打印

    int cengxu(BiTree T)
    {
    Squeue Q;
    chushihua(Q); //初始化队列
    BiTree root;
    rudui(Q, T); //树的根节点入队
    while (!(Q.front == Q.rear)) //当队列不为空时
    {

     chudui(Q, root);                   //根节点出队
 cout << root->data<<" ";
 if (root->lchild)                  //如果根节点的左孩子存在，则入队
     rudui(Q, root->lchild);
 if (root->rchild)                  //如果根节点的右孩子存在，则入队
     rudui(Q, root->rchild);

    }
    cout << endl;
    return 0;
    }
    int deep(BiTree T) //求二叉树深度
    {
    int m=0 , n=0;
    if (T ==NULL)

     return 0;

    else
    {

      m=deep(T->lchild);               //递归遍历左孩子深度
  n=deep(T->rchild);               //递归遍历右孩子深度
 if (m > n)
     return (m + 1);
 else
     return(n + 1);                //比较后输出大的作为深度

    }
    }
    int countye(BiTree T) //统计叶子节点个数
    {
    if (T == NULL) //如果树为空，退出

     return 0;

    else if (T->lchild == NULL&&T->rchild == NULL)

     return 1;                        //如果左右孩子都有，返回1

    else

     return countye(T->lchild) + countye(T->rchild);

    } //返回左右子树相加的叶子节点
    void swap(BiTree T) //交换函数
    {
    BiTree S;
    S = T->lchild;
    T->lchild = T->rchild;
    T->rchild = S;
    }
    void jiaohuan(BiTree T) //交换左右孩子
    {
    if (T == NULL)

     return ;

    else
    {

     swap(T);
 jiaohuan(T->lchild);           //中序遍历交换
 jiaohuan(T->rchild);

    }
    }
    int main()
    {
    BiTree T; //声明一个指向二叉树根节点的指针
    createBiTree(T);
    cout << “二叉树创建完成！” << endl;
    cout << “前序遍历二叉树：” << endl;
    preTraverse(T);
    cout << endl;
    cout << “中序遍历二叉树：” << endl;
    midTraverse(T);
    cout << endl;
    cout << “后序遍历二叉树：” << endl;
    postTraverse(T);
    cout << endl;
    cout << “层序遍历二叉树” << endl;
    cengxu(T);
    cout << “二叉树的深度为：”;
    cout << deep(T)<<endl;
    cout << “叶子节点的个数为：”;
    cout << countye(T)<<endl;
    jiaohuan(T);
    cout << “前序遍历交换后的二叉树：” << endl;
    preTraverse(T);
    return 0;
    }
